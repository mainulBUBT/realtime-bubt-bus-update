## 1. Overview

```
SETUP (Admin saves route - ~50 calls ever, free):
  Route → OsrmSegmentService → OSRM Route API (one call per segment)
                                        ↓
              Store in route_stops table:
                - distance_to_next_m    (road distance in meters)
                - geometry_to_next      (encoded polyline)

RUNTIME (every 5s WebSocket - ZERO API calls):
  Bus GPS → Snap to stored polyline
          → Measure distance along road to next stop
          → Show "Approaching - 1.2 km away"
```

**Same pattern as Google Maps Transit, Moovit, Transit App**

## 3. OsrmSegmentService

**New file:** `backend/app/Services/OsrmSegmentService.php`

Fetches road geometry + distance for each stop-to-stop segment using OSRM Route API.

```php
<?php

namespace App\Services;

use App\Models\Route;
use Illuminate\Support\Facades\Http;

class OsrmSegmentService
{
    public function __construct(
        private readonly RouteGeometryService $geometry,
    ) {}

    public function computeForRoute(Route $route): void
    {
        $stops = $route->stops()->orderBy('sequence')->get();
        if ($stops->count() < 2) {
            return;
        }

        $baseUrl = (string) config('services.osrm.base_url', '');

        if ($baseUrl === '') {
            $this->computeHaversineFallback($stops);
            return;
        }

        foreach ($stops as $i => $stop) {
            if ($i >= $stops->count() - 1) {
                break;
            }

            $next = $stops[$i + 1];
            $this->computeSegment($stop, $next, $baseUrl);
        }

        $stops->last()->update([
            'distance_to_next_m' => null,
            'geometry_to_next' => null,
        ]);
    }

    private function computeSegment($from, $to, string $baseUrl): void
    {
        try {
            $response = Http::timeout((float) config('services.osrm.timeout', 5))
                ->get(rtrim($baseUrl, '/') . '/route/v1/driving/' . sprintf('%.7f,%.7f;%.7f,%.7f',
                    (float) $from->lng, (float) $from->lat,
                    (float) $to->lng, (float) $to->lat
                ), [
                    'overview' => 'full',
                    'geometries' => 'polyline',
                ]);

            if (!$response->successful()) {
                $this->computeHaversineFallbackSegment($from, $to);
                return;
            }

            $data = $response->json();
            if (($data['code'] ?? '') !== 'Ok' || empty($data['routes'][0])) {
                $this->computeHaversineFallbackSegment($from, $to);
                return;
            }

            $route = $data['routes'][0];
            $from->update([
                'distance_to_next_m' => round($route['distance'], 2),
                'geometry_to_next' => $route['geometry'],
            ]);
        } catch (\Throwable) {
            $this->computeHaversineFallbackSegment($from, $to);
        }
    }

    private function computeHaversineFallback($stops): void
    {
        foreach ($stops as $i => $stop) {
            if ($i >= $stops->count() - 1) {
                break;
            }
            $this->computeHaversineFallbackSegment($stop, $stops[$i + 1]);
        }
        $stops->last()->update([
            'distance_to_next_m' => null,
            'geometry_to_next' => null,
        ]);
    }

    private function computeHaversineFallbackSegment($from, $to): void
    {
        $distance = $this->geometry->haversineMeters(
            (float) $from->lat, (float) $from->lng,
            (float) $to->lat, (float) $to->lng,
        );
        $from->update([
            'distance_to_next_m' => round($distance, 2),
            'geometry_to_next' => null,
        ]);
    }
}
```

```javascript
// Calculate distance from bus to each stop using stored geometry
// Uses existing distanceToSegmentMeters() for proper projection
function calculateRoadDistances(stops, busLat, busLng) {
  if (!busLat || !busLng || stops.length < 2) return {}

  const distances = {}

  // Find which segment the bus is on
  let busSegmentIndex = -1
  let distToNextStop = 0

  for (let i = 0; i < stops.length - 1; i++) {
    const result = distanceToSegmentMeters(
      busLat, busLng,
      parseFloat(stops[i].lat), parseFloat(stops[i].lng),
      parseFloat(stops[i + 1].lat), parseFloat(stops[i + 1].lng)
    )
    if (result.distanceMeters < 200) {
      busSegmentIndex = i
      distToNextStop = parseFloat(stops[i].distance_to_next_m || 0) * (1 - result.t)
      break
    }
  }

  // Fill distances for all stops
  for (let i = 0; i < stops.length; i++) {
    if (busSegmentIndex === -1 || i <= busSegmentIndex) {
      distances[stops[i].id] = 0  // passed or unknown
    } else if (i === busSegmentIndex + 1) {
      distances[stops[i].id] = distToNextStop  // next stop
    } else {
      // Sum segments from next stop onwards
      let d = distToNextStop
      for (let j = busSegmentIndex + 1; j < i; j++) {
        d += parseFloat(stops[j].distance_to_next_m || 0)
      }
      distances[stops[i].id] = d
    }
  }

  return distances
}
```

**Note**: `distanceToSegmentMeters()` already exists in TimelinePanel.vue and does proper point-to-line-segment projection.

## 7. Artisan Command

**New file:** `backend/app/Console/Commands/ComputeRouteDistances.php`

```php
<?php

namespace App\Console\Commands;

use App\Models\Route;
use App\Services\OsrmSegmentService;
use Illuminate\Console\Command;

class ComputeRouteDistances extends Command
{
    protected $signature = 'route:compute-distances {--route=* : Specific route IDs}';

    protected $description = 'Compute OSRM road distances and geometry for route segments';

    public function handle(OsrmSegmentService $service): int
    {
        $routes = $this->option('route')
            ? Route::whereIn('id', $this->option('route'))->get()
            : Route::with('stops')->get();

        foreach ($routes as $route) {
            if ($route->stops->count() < 2) {
                continue;
            }
            $service->computeForRoute($route);
            $this->info("Route {$route->id} ({$route->name}): computed");
        }

        return 0;
    }
}
```

Run: `php artisan route:compute-distances`
